#include<stdio.h>
int main()
{
    int a;
    printf("Enter an integer\n");
    scanf("%d",&a);
    
    while(a>0)
        {
            if(a%2==!0)
                {
                    printf("%d\n",a);
                }
            a=a-1;
        }
    
    return 0;
}
